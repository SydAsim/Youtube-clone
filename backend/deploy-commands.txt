# Run these commands to fix your Fly.io deployment:

# 1. Set essential environment variables (replace with your actual values)
fly secrets set PORT=3000
fly secrets set MONGODB_URI="your_mongodb_connection_string"
fly secrets set CORS_ORIGIN="https://your-frontend-domain.com"
fly secrets set ACCESS_TOKEN_SECRET="your_jwt_secret"
fly secrets set REFRESH_TOKEN_SECRET="your_refresh_secret"
fly secrets set CLOUDINARY_CLOUD_NAME="your_cloudinary_name"
fly secrets set CLOUDINARY_API_KEY="your_cloudinary_key"
fly secrets set CLOUDINARY_API_SECRET="your_cloudinary_secret"

# 2. Redeploy the application
fly deploy

# 3. Check the logs to see if it's working
fly logs

# 4. Test the health endpoint
curl https://youtube-backend.fly.dev/api/v1/healthcheck
